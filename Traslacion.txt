#include <GL/glut.h>

float translationX = 0.0; // Traslaci贸n en el eje X
float translationY = 0.0; // Traslaci贸n en el eje Y

void init() {
    glClearColor(1.0, 1.0, 1.0, 0.0);
    glMatrixMode(GL_PROJECTION);
    gluOrtho2D(-5.0, 5.0, -5.0, 5.0);
}

void display() {
    glClear(GL_COLOR_BUFFER_BIT);
    glColor3f(0.0, 0.0, 0.0);
    
    glLoadIdentity();
    
    // Trasladar en los ejes X e Y
    glTranslatef(translationX, translationY, 0.0);
    
     // Dibujar vectores
    glBegin(GL_QUADS);
    glVertex2f(-1.0, -1.0);
    glVertex2f(1.0, 1.0)
    glEnd();

    
    glFlush();
}

void keyboard(unsigned char key, int x, int y) {
    switch (key) {
        case 'w': // Trasladar hacia arriba
            translationY += 0.1;
            glutPostRedisplay();
            break;
        case 's': // Trasladar hacia abajo
            translationY -= 0.1;
            glutPostRedisplay();
            break;
        case 'a': // Trasladar hacia la izquierda
            translationX -= 0.1;
            glutPostRedisplay();
            break;
        case 'd': // Trasladar hacia la derecha
            translationX += 0.1;
            glutPostRedisplay();
            break;
        case 27: // Tecla ESC para salir de la aplicaci贸n
            exit(0);
            break;
        default:
            break;
    }
}

int main(int argc, char** argv) {
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_SINGLE | GLUT_RGB);
    glutInitWindowSize(400, 400);
    glutInitWindowPosition(100, 100);
    glutCreateWindow("Traslaci贸n en OpenGL");
    init();
    glutDisplayFunc(display);
    glutKeyboardFunc(keyboard);
    glutMainLoop();
    return 0;
}
